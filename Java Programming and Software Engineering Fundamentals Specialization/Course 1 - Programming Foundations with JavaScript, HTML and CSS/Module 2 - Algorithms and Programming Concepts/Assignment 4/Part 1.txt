var bgImage = new SimpleImage("drewRobert.png");
var fgImage = new SimpleImage("dinos.png");
var outputImage = new SimpleImage(bgImage.getWidth(),bgImage.getHeight());

for(var pixel of bgImage.values()){
    
    if(pixel.getGreen() > (pixel.getRed()+pixel.getBlue())){
        
        var x = pixel.getX();
        var y = pixel.getY();
        var fgImagePixel = fgImage.getPixel(x,y);
        outputImage.setPixel(x,y,fgImagePixel)
        
    }
    else{
        
        outputImage.setPixel(pixel.getX(),pixel.getY(),pixel);
        
    }

}
print(outputImage);
